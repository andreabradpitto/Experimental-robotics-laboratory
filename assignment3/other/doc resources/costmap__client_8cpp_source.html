154 geometry_msgs::Pose Costmap2DClient::getRobotPose() const
155 {
156  tf::Stamped<tf::Pose> global_pose;
157  global_pose.setIdentity();
158  tf::Stamped<tf::Pose> robot_pose;
159  robot_pose.setIdentity();
160  robot_pose.frame_id_ = robot_base_frame_;
161  robot_pose.stamp_ = ros::Time();
162  ros::Time current_time =
163  ros::Time::now(); // save time for checking tf delay later
164 
165  // get the global pose of the robot
166  try {
167  tf_->transformPose(global_frame_, robot_pose, global_pose);
168  } catch (tf::LookupException& ex) {
169  ROS_ERROR_THROTTLE(1.0, "No Transform available Error looking up robot "
170  "pose: %s\n",
171  ex.what());
172  return {};
173  } catch (tf::ConnectivityException& ex) {
174  ROS_ERROR_THROTTLE(1.0, "Connectivity Error looking up robot pose: %s\n",
175  ex.what());
176  return {};
177  } catch (tf::ExtrapolationException& ex) {
178  ROS_ERROR_THROTTLE(1.0, "Extrapolation Error looking up robot pose: %s\n",
179  ex.what());
180  return {};
181  }
182  // check global_pose timeout
183  if (current_time.toSec() - global_pose.stamp_.toSec() >
184  transform_tolerance_) {
185  ROS_WARN_THROTTLE(1.0, "Costmap2DClient transform timeout. Current time: "
186  "%.4f, global_pose stamp: %.4f, tolerance: %.4f",
187  current_time.toSec(), global_pose.stamp_.toSec(),
188  transform_tolerance_);
189  return {};
190  }
191 
192  geometry_msgs::PoseStamped msg;
193  tf::poseStampedTFToMsg(global_pose, msg);
194  return msg.pose;
195 }

197 std::array<unsigned char, 256> init_translation_table()
198 {
199  std::array<unsigned char, 256> cost_translation_table;
200 
201  // lineary mapped from [0..100] to [0..255]
202  for (size_t i = 0; i < 256; ++i) {
203  cost_translation_table[i] =
204  static_cast<unsigned char>(1 + (251 * (i - 1)) / 97);
205  }
206 
207  // special values:

210  cost_translation_table[100] = 254; // LETHAL obstacle

212 
213  return cost_translation_table;
